46template <typename Scalar>
47GreenFunctions<Scalar> construct_green_functions(
48 const std::array<typename traits::NumTraits<Scalar>::real_t, 3> &distance_vector, int order) {
49 // https://doi.org/10.1103/PhysRevA.96.062509
50 // https://doi.org/10.1103/PhysRevA.82.010901
51 // https://en.wikipedia.org/wiki/Table_of_spherical_harmonics
52
53 using real_t = typename traits::NumTraits<Scalar>::real_t;
54 constexpr real_t numerical_precision = 100 * std::numeric_limits<real_t>::epsilon();
55
56 GreenFunctions<Scalar> green_functions;
57
58 // Normalize the distance vector, return zero green functions if the distance is infinity
59 Eigen::Map<const Eigen::Vector3<real_t>> vector_map(distance_vector.data(),
60 distance_vector.size());
61 real_t distance = vector_map.norm();
62 SPDLOG_DEBUG("Interatomic distance: {}", distance);
63 if (distance == std::numeric_limits<real_t>::infinity()) {
64 return green_functions;
65 }
66 if (distance < numerical_precision) {
67 throw std::invalid_argument("The distance must be greater than zero.");
68 }
69 if (!traits::NumTraits<Scalar>::is_complex_v &&
70 std::abs(distance_vector[1]) > numerical_precision) {
71 throw std::invalid_argument(
72 "The distance vector must not have a y-component if the scalar type is real.");
73 }
74 Eigen::Vector3<real_t> vector_normalized = vector_map / distance;
75
76 // Conversion matrices from Cartesian to spherical coordinates
77 const auto &to_spherical_kappa1 = pairinteraction::spherical::CARTESIAN_TO_SPHERICAL_KAPPA1
78 .template cast<std::complex<real_t>>();
79 const auto &to_spherical_kappa2 = pairinteraction::spherical::CARTESIAN_TO_SPHERICAL_KAPPA2
80 .template cast<std::complex<real_t>>();
81
82 // Dyadic green function of dipole-dipole interaction
83 if (order >= 3) {
84 Eigen::Matrix3<real_t> green_function_cartesian = Eigen::Matrix3<real_t>::Identity() -
85 3 * vector_normalized * vector_normalized.transpose();
86
87 auto tmp = to_spherical_kappa1 * green_function_cartesian * to_spherical_kappa1.adjoint();
88 if constexpr (traits::NumTraits<Scalar>::is_complex_v) {
89 green_functions.dipole_dipole =
90 tmp.sparseView(1, numerical_precision) / std::pow(distance, 3);
91 } else {
92 green_functions.dipole_dipole =
93 tmp.real().sparseView(1, numerical_precision) / std::pow(distance, 3);
94 assert(tmp.imag().norm() < numerical_precision);
95 }
96 } else {
97 green_functions.dipole_dipole.setZero();
98 }
99
100 SPDLOG_DEBUG("Green function of dipole-dipole interaction:\n{}",
101 fmt::streamed(green_functions.dipole_dipole * std::pow(distance, 3)));
102
103 // Dyadic green function of dipole-quadrupole interaction
104 if (order >= 4) {
105 Eigen::Matrix<real_t, 3, 9> green_function_cartesian = Eigen::Matrix<real_t, 3, 9>::Zero();
106 for (Eigen::Index q = 0; q < 3; ++q) {
107 Eigen::Index row = q;
108 for (Eigen::Index j = 0; j < 3; ++j) {
109 for (Eigen::Index i = 0; i < 3; ++i) {
110 Eigen::Index col = 3 * j + i;
111 green_function_cartesian(row, col) +=
112 15 * vector_normalized[q] * vector_normalized[j] * vector_normalized[i];
113 if (i == j) {
114 green_function_cartesian(row, col) += -3 * vector_normalized[q];
115 }
116 if (i == q) {
117 green_function_cartesian(row, col) += -3 * vector_normalized[j];
118 }
119 if (j == q) {
120 green_function_cartesian(row, col) += -3 * vector_normalized[i];
121 }
122 }
123 }
124 }
125
126 auto tmp = to_spherical_kappa1 * green_function_cartesian * to_spherical_kappa2.adjoint();
127 if constexpr (traits::NumTraits<Scalar>::is_complex_v) {
128 green_functions.dipole_quadrupole =
129 tmp.sparseView(1, numerical_precision) / std::pow(distance, 4);
130 } else {
131 green_functions.dipole_quadrupole =
132 tmp.real().sparseView(1, numerical_precision) / std::pow(distance, 4);
133 assert(tmp.imag().norm() < numerical_precision);
134 }
135 } else {
136 green_functions.dipole_quadrupole.setZero();
137 }
138
139 SPDLOG_DEBUG("Green function of dipole-quadrupole interaction:\n{}",
140 fmt::streamed(green_functions.dipole_quadrupole * std::pow(distance, 4)));
141
142 // Dyadic green function of quadrupole-dipole interaction
143 if (order >= 4) {
144 Eigen::Matrix<real_t, 9, 3> green_function_cartesian = Eigen::Matrix<real_t, 9, 3>::Zero();
145 for (Eigen::Index q = 0; q < 3; ++q) {
146 for (Eigen::Index j = 0; j < 3; ++j) {
147 Eigen::Index row = 3 * q + j;
148 for (Eigen::Index i = 0; i < 3; ++i) {
149 Eigen::Index col = i;
150 green_function_cartesian(row, col) +=
151 -15 * vector_normalized[q] * vector_normalized[j] * vector_normalized[i];
152 if (i == j) {
153 green_function_cartesian(row, col) += 3 * vector_normalized[q];
154 }
155 if (i == q) {
156 green_function_cartesian(row, col) += 3 * vector_normalized[j];
157 }
158 if (j == q) {
159 green_function_cartesian(row, col) += 3 * vector_normalized[i];
160 }
161 }
162 }
163 }
164
165 auto tmp = to_spherical_kappa2 * green_function_cartesian * to_spherical_kappa1.adjoint();
166 if constexpr (traits::NumTraits<Scalar>::is_complex_v) {
167 green_functions.quadrupole_dipole =
168 tmp.sparseView(1, numerical_precision) / std::pow(distance, 4);
169 } else {
170 green_functions.quadrupole_dipole =
171 tmp.real().sparseView(1, numerical_precision) / std::pow(distance, 4);
172 assert(tmp.imag().norm() < numerical_precision);
173 }
174 } else {
175 green_functions.quadrupole_dipole.setZero();
176 }
177
178 SPDLOG_DEBUG("Green function of quadrupole-dipole interaction:\n{}",
179 fmt::streamed(green_functions.quadrupole_dipole * std::pow(distance, 4)));
180
181 // Dyadic green function of quadrupole-quadrupole interaction
182 if (order >= 5) {
183 Eigen::Matrix<real_t, 9, 9> green_function_cartesian = Eigen::Matrix<real_t, 9, 9>::Zero();
184 for (Eigen::Index q = 0; q < 3; ++q) {
185 for (Eigen::Index j = 0; j < 3; ++j) {
186 Eigen::Index row = 3 * q + j;
187 for (Eigen::Index i = 0; i < 3; ++i) {
188 for (Eigen::Index k = 0; k < 3; ++k) {
189 Eigen::Index col = 3 * i + k;
190 green_function_cartesian(row, col) += 105 * vector_normalized[q] *
191 vector_normalized[j] * vector_normalized[i] * vector_normalized[k];
192 if (i == j) {
193 green_function_cartesian(row, col) +=
194 -15 * vector_normalized[q] * vector_normalized[k];
195 }
196 if (i == q) {
197 green_function_cartesian(row, col) +=
198 -15 * vector_normalized[j] * vector_normalized[k];
199 }
200 if (j == q) {
201 green_function_cartesian(row, col) +=
202 -15 * vector_normalized[i] * vector_normalized[k];
203 }
204 if (k == q) {
205 green_function_cartesian(row, col) +=
206 -15 * vector_normalized[j] * vector_normalized[i];
207 }
208 if (k == j) {
209 green_function_cartesian(row, col) +=
210 -15 * vector_normalized[q] * vector_normalized[i];
211 }
212 if (k == i) {
213 green_function_cartesian(row, col) +=
214 -15 * vector_normalized[q] * vector_normalized[j];
215 }
216 if (q == k && i == j) {
217 green_function_cartesian(row, col) += 3;
218 }
219 if (i == k && j == q) {
220 green_function_cartesian(row, col) += 3;
221 }
222 if (j == k && i == q) {
223 green_function_cartesian(row, col) += 3;
224 }
225 }
226 }
227 }
228 }
229
230 auto tmp = to_spherical_kappa2 * green_function_cartesian * to_spherical_kappa2.adjoint();
231 if constexpr (traits::NumTraits<Scalar>::is_complex_v) {
232 green_functions.quadrupole_quadrupole =
233 tmp.sparseView(1, numerical_precision) / std::pow(distance, 5);
234 } else {
235 green_functions.quadrupole_quadrupole =
236 tmp.real().sparseView(1, numerical_precision) / std::pow(distance, 5);
237 assert(tmp.imag().norm() < numerical_precision);
238 }
239 } else {
240 green_functions.quadrupole_quadrupole.setZero();
241 }
242
243 SPDLOG_DEBUG("Green function of quadrupole-quadrupole interaction:\n{}",
244 fmt::streamed(green_functions.quadrupole_quadrupole * std::pow(distance, 5)));
245
246 return green_functions;
247}

250OperatorMatrices<Scalar>
251construct_operator_matrices(const GreenFunctions<Scalar> &green_functions,
252 const std::shared_ptr<const BasisAtom<Scalar>> &basis1,
253 const std::shared_ptr<const BasisAtom<Scalar>> &basis2) {
254 OperatorMatrices<Scalar> op;
255
256 if (green_functions.dipole_dipole.nonZeros() > 0 ||
257 green_functions.dipole_quadrupole.nonZeros() > 0) {
258 op.d1.push_back(
259 -OperatorAtom<Scalar>(basis1, OperatorType::ELECTRIC_DIPOLE, 1).get_matrix());
260 op.d1.push_back(
261 OperatorAtom<Scalar>(basis1, OperatorType::ELECTRIC_DIPOLE, 0).get_matrix());
262 op.d1.push_back(
263 -OperatorAtom<Scalar>(basis1, OperatorType::ELECTRIC_DIPOLE, -1).get_matrix());
264 }
265
266 if (green_functions.dipole_dipole.nonZeros() > 0 ||
267 green_functions.quadrupole_dipole.nonZeros() > 0) {
268 op.d2.push_back(
269 OperatorAtom<Scalar>(basis2, OperatorType::ELECTRIC_DIPOLE, -1).get_matrix());
270 op.d2.push_back(
271 OperatorAtom<Scalar>(basis2, OperatorType::ELECTRIC_DIPOLE, 0).get_matrix());
272 op.d2.push_back(
273 OperatorAtom<Scalar>(basis2, OperatorType::ELECTRIC_DIPOLE, 1).get_matrix());
274 }
275
276 if (green_functions.quadrupole_quadrupole.nonZeros() > 0 ||
277 green_functions.quadrupole_dipole.nonZeros() > 0) {
278 op.q1.push_back(
279 OperatorAtom<Scalar>(basis1, OperatorType::ELECTRIC_QUADRUPOLE, 2).get_matrix());
280 op.q1.push_back(
281 -OperatorAtom<Scalar>(basis1, OperatorType::ELECTRIC_QUADRUPOLE, 1).get_matrix());
282 op.q1.push_back(
283 OperatorAtom<Scalar>(basis1, OperatorType::ELECTRIC_QUADRUPOLE, 0).get_matrix());
284 op.q1.push_back(
285 -OperatorAtom<Scalar>(basis1, OperatorType::ELECTRIC_QUADRUPOLE, -1).get_matrix());
286 op.q1.push_back(
287 OperatorAtom<Scalar>(basis1, OperatorType::ELECTRIC_QUADRUPOLE, -2).get_matrix());
288 op.q1.push_back(
289 OperatorAtom<Scalar>(basis1, OperatorType::ELECTRIC_QUADRUPOLE_ZERO, 0).get_matrix());
290 }
291
292 if (green_functions.quadrupole_quadrupole.nonZeros() > 0 ||
293 green_functions.dipole_quadrupole.nonZeros() > 0) {
294 op.q2.push_back(
295 OperatorAtom<Scalar>(basis2, OperatorType::ELECTRIC_QUADRUPOLE, -2).get_matrix());
296 op.q2.push_back(
297 OperatorAtom<Scalar>(basis2, OperatorType::ELECTRIC_QUADRUPOLE, -1).get_matrix());
298 op.q2.push_back(
299 OperatorAtom<Scalar>(basis2, OperatorType::ELECTRIC_QUADRUPOLE, 0).get_matrix());
300 op.q2.push_back(
301 OperatorAtom<Scalar>(basis2, OperatorType::ELECTRIC_QUADRUPOLE, 1).get_matrix());
302 op.q2.push_back(
303 OperatorAtom<Scalar>(basis2, OperatorType::ELECTRIC_QUADRUPOLE, 2).get_matrix());
304 op.q2.push_back(
305 OperatorAtom<Scalar>(basis2, OperatorType::ELECTRIC_QUADRUPOLE_ZERO, 0).get_matrix());
306 }
307
308 return op;

337template <typename Scalar>
338void SystemPair<Scalar>::construct_hamiltonian() const {
339 auto basis = this->hamiltonian->get_basis();
340 auto basis1 = basis->get_basis1();
341 auto basis2 = basis->get_basis2();
342
343 auto green_functions = construct_green_functions<Scalar>(distance_vector, order);
344 auto op = construct_operator_matrices(green_functions, basis1, basis2);
345
346 // Construct the unperturbed Hamiltonian
347 this->hamiltonian = std::make_unique<OperatorPair<Scalar>>(basis, OperatorType::ENERGY);
348 this->hamiltonian_is_diagonal = true;
349 bool sort_by_quantum_number_f = basis->has_quantum_number_f();
350 bool sort_by_quantum_number_m = basis->has_quantum_number_m();
351 bool sort_by_parity = basis->has_parity();
352
353 // Dipole-dipole interaction
354 if (green_functions.dipole_dipole.nonZeros() > 0) {
355 for (Eigen::Index row = 0; row < green_functions.dipole_dipole.rows(); ++row) {
356 for (typename Eigen::SparseMatrix<Scalar, Eigen::RowMajor>::InnerIterator it(
357 green_functions.dipole_dipole, row);
358 it; ++it) {
359 this->hamiltonian->get_matrix() += it.value() *
360 utils::calculate_tensor_product(basis, basis, op.d1[it.row()], op.d2[it.col()]);
361 if (it.row() != it.col()) {
362 sort_by_quantum_number_m = false;
363 }
364 }
365 }
366 this->hamiltonian_is_diagonal = false;
367 sort_by_quantum_number_f = false;
368 }
369
370 // Dipole-quadrupole interaction
371 if (green_functions.dipole_quadrupole.nonZeros() > 0) {
372 for (Eigen::Index row = 0; row < green_functions.dipole_quadrupole.rows(); ++row) {
373 for (typename Eigen::SparseMatrix<Scalar, Eigen::RowMajor>::InnerIterator it(
374 green_functions.dipole_quadrupole, row);
375 it; ++it) {
376 this->hamiltonian->get_matrix() += it.value() *
377 utils::calculate_tensor_product(basis, basis, op.d1[it.row()], op.q2[it.col()]);
378 if (it.row() != it.col() - 1) {
379 sort_by_quantum_number_m = false;
380 }
381 }
382 }
383 this->hamiltonian_is_diagonal = false;
384 sort_by_quantum_number_f = false;
385 }
386
387 // Quadrupole-dipole interaction
388 if (green_functions.quadrupole_dipole.nonZeros() > 0) {
389 for (Eigen::Index row = 0; row < green_functions.quadrupole_dipole.rows(); ++row) {
390 for (typename Eigen::SparseMatrix<Scalar, Eigen::RowMajor>::InnerIterator it(
391 green_functions.quadrupole_dipole, row);
392 it; ++it) {
393 this->hamiltonian->get_matrix() += it.value() *
394 utils::calculate_tensor_product(basis, basis, op.q1[it.row()], op.d2[it.col()]);
395 if (it.row() - 1 != it.col()) {
396 sort_by_quantum_number_m = false;
397 }
398 }
399 }
400 this->hamiltonian_is_diagonal = false;
401 sort_by_quantum_number_f = false;
402 }
403
404 // Quadrupole-quadrupole interaction
405 if (green_functions.quadrupole_quadrupole.nonZeros() > 0) {
406 for (Eigen::Index row = 0; row < green_functions.quadrupole_quadrupole.rows(); ++row) {
407 for (typename Eigen::SparseMatrix<Scalar, Eigen::RowMajor>::InnerIterator it(
408 green_functions.quadrupole_quadrupole, row);
409 it; ++it) {
410 this->hamiltonian->get_matrix() += it.value() *
411 utils::calculate_tensor_product(basis, basis, op.q1[it.row()], op.q2[it.col()]);
412 if (it.row() != it.col()) {
413 sort_by_quantum_number_m = false;
414 }
415 }
416 }
417 this->hamiltonian_is_diagonal = false;
418 sort_by_quantum_number_f = false;
419 }
420
421 // Store which labels can be used to block-diagonalize the Hamiltonian
422 this->blockdiagonalizing_labels.clear();
423 if (sort_by_quantum_number_f) {
424 this->blockdiagonalizing_labels.push_back(TransformationType::SORT_BY_QUANTUM_NUMBER_F);
425 }
426 if (sort_by_quantum_number_m) {
427 this->blockdiagonalizing_labels.push_back(TransformationType::SORT_BY_QUANTUM_NUMBER_M);
428 }
429 if (sort_by_parity) {
430 this->blockdiagonalizing_labels.push_back(TransformationType::SORT_BY_PARITY);
431 }
432}
