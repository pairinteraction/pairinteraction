52 // Add external fields (see https://arxiv.org/abs/1612.08053 for details)
53
54 // Stark effect: - \vec{d} \vec{E} = - d_{1,0} E_{0} + d_{1,1} E_{-} + d_{1,-1} E_{+}
55 // with the electric dipole operator: d_{1,q} = - e r sqrt{4 pi / 3} Y_{1,q}(\theta, \phi)
56 // where electric_field_spherical=[E_{-}, E_{0}, E_{+}]
57 if (std::abs(electric_field_spherical[1]) > numerical_precision) {
58 *this->hamiltonian -= electric_field_spherical[1] *
59 OperatorAtom<Scalar>(basis, OperatorType::ELECTRIC_DIPOLE, 0);
60 this->hamiltonian_is_diagonal = false;
61 sort_by_quantum_number_f = false;
62 sort_by_parity = false;
63 }
64 if (std::abs(electric_field_spherical[2]) > numerical_precision) {
65 *this->hamiltonian += electric_field_spherical[2] *
66 OperatorAtom<Scalar>(basis, OperatorType::ELECTRIC_DIPOLE, -1);
67 this->hamiltonian_is_diagonal = false;
68 sort_by_quantum_number_f = false;
69 sort_by_quantum_number_m = false;
70 sort_by_parity = false;
71 }
72 if (std::abs(electric_field_spherical[0]) > numerical_precision) {
73 *this->hamiltonian += electric_field_spherical[0] *
74 OperatorAtom<Scalar>(basis, OperatorType::ELECTRIC_DIPOLE, 1);
75 this->hamiltonian_is_diagonal = false;
76 sort_by_quantum_number_f = false;
77 sort_by_quantum_number_m = false;
78 sort_by_parity = false;
79 }
80
81 // Zeeman effect: - \vec{\mu} \vec{B} = - \mu_{1,0} B_{0} + \mu_{1,1} B_{-} + \mu_{1,-1} B_{+}
82 // with the magnetic dipole operator: \vec{\mu} = - \mu_B / \hbar (g_l \vec{l} + g_s \vec{s})
83 // where magnetic_field_spherical=[B_{-}, B_{0}, B_{+}]
84 if (std::abs(magnetic_field_spherical[1]) > numerical_precision) {
85 *this->hamiltonian -= magnetic_field_spherical[1] *
86 OperatorAtom<Scalar>(basis, OperatorType::MAGNETIC_DIPOLE, 0);
87 this->hamiltonian_is_diagonal = false;
88 sort_by_quantum_number_f = false;
89 }
90 if (std::abs(magnetic_field_spherical[2]) > numerical_precision) {
91 *this->hamiltonian += magnetic_field_spherical[2] *
92 OperatorAtom<Scalar>(basis, OperatorType::MAGNETIC_DIPOLE, -1);
93 this->hamiltonian_is_diagonal = false;
94 sort_by_quantum_number_f = false;
95 sort_by_quantum_number_m = false;
96 }
97 if (std::abs(magnetic_field_spherical[0]) > numerical_precision) {
98 *this->hamiltonian += magnetic_field_spherical[0] *
99 OperatorAtom<Scalar>(basis, OperatorType::MAGNETIC_DIPOLE, 1);
100 this->hamiltonian_is_diagonal = false;
101 sort_by_quantum_number_f = false;
102 sort_by_quantum_number_m = false;
103 }
104
105 // Diamagnetism: 1 / (8 m_e) abs(\vec{d} \times \vec{B})^2
106 // = (1/12) \left[ B_0^2 (q0 - d_{2,0}) + B_+ B_- (-2 q0 - d_{2,0})
107 // + \sqrt{3} B_0 B_- d_{2,1} + \sqrt{3} B_0 B_+ d_{2,-1}
108 // - \sqrt{3/2} B_-^2 d_{2,2} - \sqrt{3/2} B_+^2 d_{2,-2} \right]
109 // with the operator: q0 = e^2 r^2 sqrt{4 pi} Y_{0,0} = e^2 r^2
110 // and the electric quadrupole operator: d_{2,q} = e^2 r^2 sqrt{4 pi / 5} Y_{2,q}(\theta, \phi)
111 if (diamagnetism_enabled) {
112 if (std::abs(magnetic_field_spherical[1]) > numerical_precision) {
113 *this->hamiltonian += static_cast<real_t>(1 / 12.) *
114 static_cast<Scalar>(std::pow(magnetic_field_spherical[1], 2)) *
115 OperatorAtom<Scalar>(basis, OperatorType::ELECTRIC_QUADRUPOLE_ZERO, 0);
116 *this->hamiltonian -= static_cast<real_t>(1 / 12.) *
117 static_cast<Scalar>(std::pow(magnetic_field_spherical[1], 2)) *
118 OperatorAtom<Scalar>(basis, OperatorType::ELECTRIC_QUADRUPOLE, 0);
119 this->hamiltonian_is_diagonal = false;
120 sort_by_quantum_number_f = false;
121 }
122 if (std::abs(magnetic_field_spherical[2]) > numerical_precision &&
123 std::abs(magnetic_field_spherical[0]) > numerical_precision) {
124 *this->hamiltonian -= static_cast<real_t>(2 / 12.) * magnetic_field_spherical[2] *
125 magnetic_field_spherical[0] *
126 OperatorAtom<Scalar>(basis, OperatorType::ELECTRIC_QUADRUPOLE_ZERO, 0);
127 *this->hamiltonian -= static_cast<real_t>(1 / 12.) * magnetic_field_spherical[2] *
128 magnetic_field_spherical[0] *
129 OperatorAtom<Scalar>(basis, OperatorType::ELECTRIC_QUADRUPOLE, 0);
130 this->hamiltonian_is_diagonal = false;
131 sort_by_quantum_number_f = false;
132 }

143 std::abs(magnetic_field_spherical[2]) > numerical_precision) {
144 *this->hamiltonian += static_cast<real_t>(std::sqrt(3.0) / 12.) *
145 magnetic_field_spherical[1] * magnetic_field_spherical[2] *
146 OperatorAtom<Scalar>(basis, OperatorType::ELECTRIC_QUADRUPOLE, -1);

165 sort_by_quantum_number_m = false;

170 this->blockdiagonalizing_labels.clear();
171 if (sort_by_quantum_number_f) {
172 this->blockdiagonalizing_labels.push_back(TransformationType::SORT_BY_QUANTUM_NUMBER_F);
