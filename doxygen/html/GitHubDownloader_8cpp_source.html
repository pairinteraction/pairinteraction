22GitHubDownloader::download(const std::string &remote_url, const std::string &if_modified_since,
23 bool use_octet_stream) const {
24 return std::async(
25 std::launch::async, [this, remote_url, if_modified_since, use_octet_stream]() -> Result {
26 // Prepare headers
27 httplib::Headers headers{
28 {"X-GitHub-Api-Version", "2022-11-28"},
29 {"Accept",
30 use_octet_stream ? "application/octet-stream" : "application/vnd.github+json"}};
31
32 if (!if_modified_since.empty()) {
33 headers.emplace("if-modified-since", if_modified_since);
34 }
35
36 // Use the GitHub token if available; otherwise, if we have a conditional request,
37 // insert a dummy authorization header to avoid increasing rate limits
38 if (auto *token = std::getenv("GITHUB_TOKEN"); token) {
39 headers.emplace("Authorization", fmt::format("Bearer {}", token));
40 } else if (!if_modified_since.empty()) {
41 headers.emplace("Authorization",
42 "avoids-an-increase-in-ratelimits-used-if-304-is-returned");
43 }
44
45 auto response = client->Get(remote_url, headers);
46
47 // Handle if the response is null
48 if (!response) {
49 // Defensive handling: if response is null and the error is unknown,
50 // treat this as a 304 Not Modified
51 if (response.error() == httplib::Error::Unknown) {
52 return Result{304, "", "", {}};
53 }
54 throw std::runtime_error(fmt::format("Error downloading '{}': {}", remote_url,
55 httplib::to_string(response.error())));
56 }
57
58 // Parse the response
59 Result result;
60 if (response->has_header("x-ratelimit-remaining")) {
61 result.rate_limit.remaining =
62 std::stoi(response->get_header_value("x-ratelimit-remaining"));
63 }
64 if (response->has_header("x-ratelimit-reset")) {
65 result.rate_limit.reset_time =
66 std::stoi(response->get_header_value("x-ratelimit-reset"));
67 }
68 if (response->has_header("last-modified")) {
69 result.last_modified = response->get_header_value("last-modified");
70 }
71 result.body = response->body;
72 result.status_code = response->status;
73 return result;
74 });
75}
