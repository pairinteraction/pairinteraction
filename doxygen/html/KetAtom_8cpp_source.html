37std::string KetAtom::get_label() const {
38 size_t pos = species.find('_');
39 std::string label = (pos != std::string::npos) ? species.substr(0, pos) : species;
40 label[0] = static_cast<char>(std::toupper(label[0]));
41
42 if (!is_calculated_with_mqdt) {
43 if (quantum_number_s_exp == 0) {
44 label += "_singlet";
45 } else if (quantum_number_s_exp == 1) {
46 label += "_triplet";
47 } else if (quantum_number_s_exp != 0.5) {
48 throw std::runtime_error(
49 "Invalid value for quantum number s in the single-channel description.");
50 }
51 }
52
53 label += ":";
54
55 if (is_calculated_with_mqdt) {
56 label += fmt::format("S={:.1f},nu={:.1f},L={:.1f},", quantum_number_s_exp,
57 quantum_number_nu, quantum_number_l_exp);
58 label += this->is_j_total_momentum ? "J=" : "F=";
59 } else {
60 label += fmt::format("{:d},", quantum_number_n);
61 if (quantum_number_l_exp == std::rint(quantum_number_l_exp) &&
62 quantum_number_l_exp < quantum_number_l_labels.size()) {
63 label += quantum_number_l_labels.at(static_cast<size_t>(quantum_number_l_exp));
64 } else {
65 label += fmt::format("{:.0f}", quantum_number_l_exp);
66 }
67 label += "_";
68 }
69
70 double total_momentum =
71 this->is_j_total_momentum ? this->quantum_number_j_exp : this->quantum_number_f;
72 if (total_momentum == std::rint(total_momentum)) {
73 label += fmt::format("{:.0f}", total_momentum);
74 } else if (2 * total_momentum == std::rint(2 * total_momentum)) {
75 label += fmt::format("{:.0f}/2", 2 * total_momentum);
76 } else {
77 std::abort(); // can't happen because the total momentum is validated to be an integer
78 // or half-integer
79 }
80
81 if (this->quantum_number_m == std::rint(this->quantum_number_m)) {
82 label += fmt::format(",{:.0f}", this->quantum_number_m);
83 } else if (2 * this->quantum_number_m == std::rint(2 * this->quantum_number_m)) {
84 label += fmt::format(",{:.0f}/2", 2 * this->quantum_number_m);
85 } else {
86 std::abort(); // can't happen because the quantum number m is validated to be an integer
87 // or half-integer
88 }
89
90 return label;
91}
